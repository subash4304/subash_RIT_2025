def findPoint(px, py, qx, qy):
    rx = 2 * qx - px
    ry = 2 * qy - py
    return rx, ry

def main():
    n = int(input())
    for _ in range(n):
        px, py, qx, qy = map(int, input().split())
        reflected_point = findPoint(px, py, qx, qy)
        print(*reflected_point)
if __name__ == "__main__":
    main()
